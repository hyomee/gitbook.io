# 2-4. 객체와 프로타입 차이

## 1. 객체

```javascript
function Animal(name) {
    this.name = name;
    this.getName = function () {
        console.log(this.name);
    }
}
var monkey = new Animal("Monkey");
var dog = new Animal("dog");
var cat = new Animal("cat");
monkey.getName();           // Monkey
dog.getName();              // dog
cat.getName();              // cat

```

## 2. 프로타입

```javascript
Object.prototype.getName = function () {
    console.log(this.name);
}
function Animal(name) {
    this.name = name;
}
var monkey = new Animal("Monkey");
var dog = new Animal("dog");
var cat = new Animal("cat");
monkey.getName();       // Monkey
dog.getName();          // dog
cat.getName();          // cat
```
